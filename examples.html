  $(document).ready(function() {
    $('#responsive').lightSlider({
        item:4,
        slideMove:2,
        easing: 'cubic-bezier(0.25, 0, 0.25, 1)',
        speed:600,
        responsive : [
            {
                breakpoint:800,
                settings: {
                    item:3,
                    slideMove:1,
                    slideMargin:6,
                  }
            },
            {
                breakpoint:480,
                settings: {
                    item:2,
                    slideMove:1
                  }
            }
        ]
    });  
  });

Note

  • You can add as many breakpoints as you want.
  • supports item, slideMove, slideMargin, thumbItem, thumbMargin

Public Methods

  $(document).ready(function() {
    var slider = $('#publicMethods').lightSlider({
        slideMargin:4,
        slideWidth:200,
        loop:false
    });
    $('#goToSlide').click(function(){
        slider.goToSlide(3);    
    });
    $('#goToPrevSlide').click(function(){
        slider.goToPrevSlide(); 
    });
    $('#goToNextSlide').click(function(){
        slider.goToNextSlide(); 
    });
    $('#getCurrentSlideCount').click(function(){
        slider.getCurrentSlideCount(); 
    });
    $('#addSlide').click(function() {
        var newEl = ' <li> <a href="javascript:void(0)"><img src="img/cS-1.jpg" /></a> </li>';
        $('#publicMethods').prepend(newEl);
        slider.refresh();
    });
    $('#play').click(function(){
        slider.play();    
    });
    $('#pause').click(function(){
        slider.pause();    
    });
  });

$(document).ready(function() {
  $('#callback').lightSlider({
    onBeforeStart: function (plugin) {
        console.log('onBeforeStart',plugin);
    },
    onSliderLoad: function (plugin) {
        plugin.addClass('classAddedBeforeSliderLoad');
    },
    onBeforeSlide: function (plugin,scene) {
        alert('onBeforeSlide');
        console.log(scene);
    },
    onAfterSlide: function (plugin,scene) {
        alert('onAfterSlide');
        console.log(scene);
    },
    onBeforeNextSlide: function (plugin,scene) {
        alert('onBeforeNextSlide');
        console.log(scene);
    },
    onBeforePrevSlide: function (plugin,scene) {
        alert('onBeforePrevSlide');
        console.log(scene);
    }
  });
});
